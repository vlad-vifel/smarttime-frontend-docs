const roles = {
  Администратор: "А",

/**
 * Функция получения доступных для указанной роли муршрутов
 * @method
 * @param {string} role Роль пользователя
 * @returns {array} Массив объектов, описывающих доступные маршруты
 */
export const getRoutes = (role) => {
  role = roles[role];

  if (!(role == "А" || role == "УО" || role == "П" || role == "МД" || role == "С")) {
    return [
      {
        path: "/",
        exact: true,
        Component: Outlet,
        errorElement: Outlet,
        children: [
          {
            path: "login",
            Component: Login,
          },
          {
            path: "",
            Component: RedirectToLogin,
          },
          {
            path: "*",
            Component: RedirectToLogin,
          },
        ],
      },
    ]
  }

  const routes = [
    {
      path: "/",
      exact: true,
      Component: Root,
      errorElement: Root,
      children: [
        {
          path: "",
          Component: Home,
        },
        {
          path: "help",
          Component: Help,
        },
        {
          path: "instruction",
          Component: Instruction,
        },
        {
          path: "login",
          Component: Login,
        },
        {
          path: "*",
          Component: Page404,
        },
      ],
    },
  ];

  // Расписание сессий
  if (role === "А" || role === "УО" || role === "П") {
    
    const tmp = {
      path: "schedule",
      Component: Outlet,
      children: [
        {
          path: "",
          Component: Schedule,
        },
        {
          path: "2023-4",
          Component: EditSchedule,
        },
        {
          path: "2023-3",
          Component: EditSchedule,
        },
        {
          path: "2023-2",
          Component: EditSchedule,
        },
        {
          path: "2023-1",
          Component: EditSchedule,
        },
      ],
    };
    routes[0].children.push(tmp);
  }

  // Анкетирование
  if (role === "А" || role === "УО" || role === "П") {
    const tmp = {
      path: "polling",
      Component: Outlet,
      children: [
        {
          path: "",
          Component: Polling,
        },
        {
          path: "form",
          Component: Outlet,
          children: [
            {
              path: "",
              Component: Page404,
            },
            {
              path: "2023-4",
              Component: Form,
            },
            {
              path: "2023-3",
              Component: Form,
            },
            {
              path: "2023-2",
              Component: Form,
            },
            {
              path: "2023-1",
              Component: Form,
            },
          ],
        },
      ],
    };

    if (role === "А" || role === "УО") {
      tmp.children.push({
        path: "results",
        Component: Outlet,
        children: [
          {
            path: "",
            Component: Page404,
          },
          {
            path: "2023-4",
            Component: Results,
          },
          {
            path: "2023-3",
            Component: Results,
          },
          {
            path: "2023-2",
            Component: Results,
          },
          {
            path: "2023-1",
            Component: Results,
          },
        ],
      });
    }
    routes[0].children.push(tmp);
  }

  // Пользователи
  if (role === "А") {
    const tmp = {
      path: "users",
      Component: Users,
    };
    routes[0].children.push(tmp);
  }

  // Справочники
  if (role === "А" || role === "УО" || role === "МД") {
    const tmp = {
      path: "guides",
      Component: Outlet,
      children: [
        {
          path: "",
          Component: Guides,
        },
        {
          path: "workload",
          Component: Workload,
        },
        {
          path: "auditorium",
          Component: Auditorium,
        },
      ],
    };
    routes[0].children.push(tmp);
  }

  return routes
};

/**
 * Функция получения доступных для указанной роли ссылок
 * @method
 * @param {string} role Роль пользователя
 * @returns {array} Массив объектов, описывающих доступные ссылки
 */
export const getLinks = (role) => {
  role = roles[role]

  if (!(role == "А" || role == "УО" || role == "П" || role == "МД" || role == "С")) {
    return []
  }

  const links = [
    {
      id: 0,
      title: "Главная",
      to: "/",
      icon: HomeIcon,
    },
    {
      id: 5,
      title: "Помощь",
      to: "/help",
      icon: HelpIcon,
    },
    {
      id: 6,
      title: "Инструкция",
      to: "/instruction",
      icon: InstructionIcon,
    },
  ]

  // Расписание сессий
  if (role === "А" || role === "УО" || role === "П") {
    links.push(
      {
        id: 1,
        title: "Расписание сессии",
        to: "/schedule",
        icon: ScheduleIcon,
      }
    )
  }

  // Анкетирование
  if (role === "А" || role === "УО" || role === "П") {
    links.push(
      {
        id: 2,
        title: "Анкетирование",
        to: "/polling",
        icon: PollingIcon,
      }
    )
  }

  // Пользователи
  if (role === "А") {
    links.push(
      {
        id: 3,
        title: "Пользователи",
        to: "/users",
        icon: UsersIcon,
      }
    )
  }

  // Справочники
  if (role === "А" || role === "УО" || role === "МД") {
    links.push(
      {
        id: 4,
        title: "Справочники",
        to: "/guides",
        icon: GuidesIcon,
      }
    )
  }

  return links.sort((a, b) => a.id - b.id);
}
