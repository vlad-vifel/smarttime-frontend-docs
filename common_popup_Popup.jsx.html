import styles from "./Popup.module.css";
import { useRef } from "react";
import PropTypes from "prop-types";
import { createPortal } from "react-dom";
import classNames from "classnames";
import useOutsideMousedown from "../../../hooks/UseOutsideMouseDown";

/**
 * Компонент всплывающего окна
 * @component
 */
const Popup = ({ headerTitle, children, isActive, setActive, className }) => {
  const popupRef = useRef(null);

  /**
   * Функция закрытия всплывающего окна
   */
  const close = () => {
    setActive(false);
  };

  useOutsideMousedown(popupRef, close);
  
  return (
    <div>
      {isActive &&
        createPortal(
          <div className={styles.popup_shadow}>
            <div ref={popupRef} className={styles.popup_alert_main}>
              {/* Заголовок окна */}
              <div className={styles.popup_alert_header}>{headerTitle}</div>
              {/* Содержимое окна */}
              <div className={classNames(styles.popup_alert_content, className)}>{children}</div>
            </div>
          </div>,
          document.body
        )}
    </div>
  );
};

Popup.propTypes = {
  /**
   * Заголовок окна
   */
  headerTitle: PropTypes.string.isRequired,
  /**
   * Содержимое окна
   */
  children: PropTypes.node.isRequired,
  /**
   * Флаг видимости окна
   */
  isActive: PropTypes.bool.isRequired,
  /**
   * Функция установки видимости окна
   */
  setActive: PropTypes.func.isRequired,
  /**
   * Дополнительный класс для стилизации
   */
  className: PropTypes.string,
};

export default Popup;
