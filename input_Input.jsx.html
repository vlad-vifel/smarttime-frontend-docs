import { useDispatch } from "react-redux";
import styles from "./Input.module.css";
import { updateCell } from "../../../store/pollingSlice";
import { useState } from "react";
import PropTypes from "prop-types";

/**
 * Компонент поля ввода текста
 *
 * @component
 */
const Input = ({ value, c, name, hint, isUpdate }) => {
  const [text, setText] = useState(value);
  const dispatcher = useDispatch();

  const saveText = (e) => {
    setText(e.target.value);
    isUpdate &&
      dispatcher(
        updateCell({
          column: c,
          row: name,
          value: e.target.value,
        })
      );
  };

  return (
    <div className={styles.input_container}>
      <input placeholder={hint} onChange={saveText} value={text} />
    </div>
  );
};

Input.propTypes = {
  /**
   * Текст по умолчанию
   */
  value: PropTypes.string,
  /**
   * Название колонки
   */
  с: PropTypes.string,
  /**
   * Название ряда
   */
  name: PropTypes.string,
  /**
   * Подсказка
   */
  hint: PropTypes.string,
  /**
   * Флаг, показывающий, надо ли обновлять информацию в redux хранилище
   */
  isUpdate: PropTypes.bool,
};

export default Input;
