import { useContext } from "react";
import classNames from "classnames";
import PropTypes from "prop-types";
import { DnDContext } from "react-big-calendar/lib/addons/dragAndDrop/DnDContext";
import { accessor as getAccessor } from "react-big-calendar/lib/utils/accessors";
import "./MyEventWrapperComponent.css";

const Directions = {
  up: "up",
  down: "down",
  left: "left",
  right: "right",
};

/**
 * Функция рассчета ширины карточки
 * @param {number} width Ширина карточки
 * @param {number} left Отсутп слева
 * @returns {number} Итоговая ирина карточки
 */
const calcWidth = (width, left) => {
  return width + left > 99.9 ? width : width / 1.7;
};

/**
 * Компонент карточки экзамена
 * @component
 */
const MyEventWrapperComponentDraggable = (props) => {
  const {
    style,
    onClick,
    event,
    resource,
    type,
    continuesPrior,
    continuesAfter,
    resizable,
    className,
  } = props;

  const wrapperStyles = {
    ...(style?.height ? { height: `${style.height}%` } : {}),
    ...(style?.width
      ? { width: `${calcWidth(style.width, style.xOffset || 0)}%` }
      : {}),
    ...(style?.top ? { top: `${style.top}%` } : {}),
    ...(style?.xOffset ? { left: `${style.xOffset}%` } : {}),
  };

  const context = useContext(DnDContext);

  /**
   * Функция обработки изменения размера карточки вверх
   * @param {Event} e Событие изменения размера карточки
   */
  const handleResizeUp = (e) => {
    if (e.button !== 0) {
      return;
    }
    context.draggable.onBeginAction(event, "resize", "UP");
  };

  /**
   * Функция обработки изменения размера карточки вниз
   * @param {Event} e Событие изменения размера карточки
   */
  const handleResizeDown = (e) => {
    if (e.button !== 0) {
      return;
    }
    context.draggable.onBeginAction(event, "resize", "DOWN");
  };

  /**
   * Функция обработки изменения размера карточки влево
   * @param {Event} e Событие изменения размера карточки
   */
  const handleResizeLeft = (e) => {
    if (e.button !== 0) {
      return;
    }
    context.draggable.onBeginAction(event, "resize", "LEFT");
  };

  /**
   * Функция обработки изменения размера карточки вправо
   * @param {Event} e Событие изменения размера карточки
   */
  const handleResizeRight = (e) => {
    if (e.button !== 0) {
      return;
    }
    context.draggable.onBeginAction(event, "resize", "RIGHT");
  };

  /**
   * Функция рендера ползунка для изменения размера карточки по направлению
   * @param {string} direction направление изменения карточки
   * @returns {JSX.Element} Ползунок для изменения размера карточки
   */
  const renderAnchor = (direction) => {
    const cls =
      direction === Directions.up || direction === Directions.down
        ? "ns"
        : "ew";

    /**
     * Функция выбора функции обработки изменения карточки по направлению
     * @param {string} direction направление изменения карточки
     */
    const selectHandler = (direction) => {
      switch (direction) {
        case Directions.up:
          return handleResizeUp;
        case Directions.down:
          return handleResizeDown;
        case Directions.left:
          return handleResizeLeft;
        case Directions.right:
          return handleResizeRight;
        default:
          return null;
      }
    };

    return (
      <div
        className={`rbc-addons-dnd-resize-${cls}-anchor`}
        onMouseDown={selectHandler(direction)}
      >
        {/* можно вставить любую иконку в зависимости от переменной cls */}
        <div className={`rbc-addons-dnd-resize-${cls}-icon`} />
      </div>
    );
  };

  /**
   * Функция рендера карточки экзамена
   * @param {object} wrapperProps Свойства компонента-обертки
   * @param {object} anchors Ползунки карточки
   * @returns {JSX.Element} Карточка экзамена
   */
  const render = (wrapperProps = {}, anchors) => {
    return (
      <div
        role="button"
        style={wrapperStyles}
        onClick={onClick}
        {...wrapperProps}
        className={classNames(
          "rbc-event",
          "my-event-wrapper",
          wrapperProps.className
        )}
      >
        {anchors && anchors.startAnchor}
        <props.components.event {...props} />
        {anchors && anchors.endAnchor}
      </div>
    );
  };

  const handleStartDragging = (e) => {
    if (e.button !== 0) {
      return;
    }

    const isResizeHandle = e.target
      .getAttribute("class")
      ?.includes("rbc-addons-dnd-resize");
    if (!isResizeHandle) {
      let extendedEvent = { ...event };
      extendedEvent.sourceResource = resource;
      context.draggable.onBeginAction(event, "move");
    }
  };

  if (className === "rbc-addons-dnd-drag-preview") {
    const newProps = {
      className,
    };
    return render(newProps);
  }

  const { draggable } = context;
  const { draggableAccessor, resizableAccessor } = draggable;

  const isDraggable = draggableAccessor
    ? !!getAccessor(event, draggableAccessor)
    : true;

  if (!isDraggable) {
    return render();
  }

  const isResizable =
    resizable &&
    (resizableAccessor ? !!getAccessor(event, resizableAccessor) : true);

  const newProps = {};
  const anchors = {};

  if (isResizable || isDraggable) {
    newProps.onMouseDown = handleStartDragging;
    newProps.onTouchStart = handleStartDragging;

    if (isResizable) {
      let StartAnchor = null;
      let EndAnchor = null;

      if (type === "date") {
        StartAnchor = !continuesPrior && renderAnchor(Directions.left);
        EndAnchor = !continuesAfter && renderAnchor(Directions.right);
      } else {
        StartAnchor = !continuesPrior && renderAnchor(Directions.up);
        EndAnchor = !continuesAfter && renderAnchor(Directions.down);
      }

      anchors.startAnchor = StartAnchor;
      anchors.endAnchor = EndAnchor;
    }

    if (
      draggable.dragAndDropAction.interacting &&
      draggable.dragAndDropAction.event === event
    ) {
      newProps.className = "rbc-addons-dnd-dragged-event";
    }
  }

  return render(newProps, anchors);
};

MyEventWrapperComponentDraggable.propTypes = {
  /**
   * Дополнительный класс для стилизации
   */
  className: PropTypes.string,
  /**
   * Стили карточки экзамена
   */
  style: PropTypes.shape({
    height: PropTypes.number,
    width: PropTypes.number,
    top: PropTypes.number,
    xOffset: PropTypes.number,
  }),
  /**
   * Функция нажатия на карточку
   */
  onClick: PropTypes.func,
  /**
   * Свойства экзамена
   */
  event: PropTypes.object.isRequired,
  /**
   * Содержимое экзамена
   */
  children: PropTypes.node.isRequired,
  /**
   * Тип экзамена
   */
  type: PropTypes.oneOf(["date", "time"]),
  /**
   * Флаг, показывающий, можно ли перемещать карточку
   */
  draggable: PropTypes.bool,
  /**
   * Флаг, показывающий, идет ли экзамен весь день
   */
  allDay: PropTypes.bool,
  /**
   * ...
   */
  isRow: PropTypes.bool,
  /**
   * ...
   */
  continuesPrior: PropTypes.bool,
  /**
   * ...
   */
  continuesAfter: PropTypes.bool,
  /**
   * Флаг, показывающий, перемещается ли данный экзамен в этот момент
   */
  isDragging: PropTypes.bool,
  /**
   * Флаг, показывающий, меняется ли размер данного экзамена  в этот момент
   */
  isResizing: PropTypes.bool,
  /**
   * ...
   */
  resource: PropTypes.number,
  /**
   * Флаг, показывающий, можно ли менять размер карточки
   */
  resizable: PropTypes.bool,
};

export default MyEventWrapperComponentDraggable;
