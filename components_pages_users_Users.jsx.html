import StaticPage from "../common/staticPage/StaticPage";
import styles from "./Users.module.css";
import global_styles from "../../../styles/global.module.css";
import classNames from "classnames";
import Table from "../../common/table/Table";
import { useDispatch, useSelector } from "react-redux";
import { toggleRemoveAllFilters } from "../../../store/globalSlice";
import DownloadTable from "../../common/downloadTable/DownloadTable";
import UserCellDropdown from "./userCellDropdown/UserCellDropdown";
import { useEffect, useState } from "react";
import { getUsers } from "./requests";
import TokenPopup from "./tokenPopup/TokenPopup";
import { ThreeDots } from "react-loader-spinner";

/**
 * Функция сортировки для ссылочных столбцов
 * @param {object} rowA Объект строки A
 * @param {object} rowB Объект строки B
 * @param {string} id Идентификатор столбца
 * @param {boolean} desc Флаг сортировки по убыванию
 * @returns {number} Результат сравнения
 */
const linkSort = (rowA, rowB, id, desc) => {
  let a, b;
  if (typeof rowA.values[id] === "string") {
    a = `${rowA.values[id]}`;
  } else if (
    rowA.values[id].props &&
    typeof rowA.values[id].props.children === "string"
  ) {
    a = `${rowA.values[id].props.children}`;
  }

  if (typeof rowB.values[id] === "string") {
    b = `${rowB.values[id]}`;
  } else if (
    rowB.values[id].props &&
    typeof rowB.values[id].props.children === "string"
  ) {
    b = `${rowB.values[id].props.children}`;
  }
  if (a > b) return 1;
  else if (a < b) return -1;
  else return 0;
};

const columns = [
  {
    Header: "Фамилия Имя Отчество",
    accessor: "fio",
    width: "450",
    sortType: linkSort,
  },
  {
    Header: "Рабочий email",
    accessor: "email",
    width: "280",
    sortType: "string",
  },
  {
    Header: "Роль",
    accessor: "roles",
    width: "400",
    disableSortBy: true,
    disableFilters: true,
  },
  {
    Header: "Дата авторизации",
    accessor: "authorization_date",
    width: "240",
    sortType: "date",
  },
  {
    Header: "Состояние",
    accessor: "last_login_date",
    width: "240",
    sortType: "date",
  },
  {
    Header: "Авторизация",
    accessor: "token",
    width: "auto",
    disableSortBy: true,
    disableFilters: true,
  },
];

/**
 * Компонент страницы Пользователи
 * @component
 */
const Users = () => {
  const [data, setData] = useState([]);
  const dispatcher = useDispatch();

  const hiddenColumns = useSelector((state) => state.workload.hiddenColumns);

  useEffect(() => {
    getUsers().then((data) => setData(data));
  }, []);

  data.forEach((row, i) => {
    if (row["info"] && typeof row["fio"] === "string") {
      row["fio"] = (
        <a href={row["info"]} className={styles.user_link} target="_blank">
          {row["fio"]}
        </a>
      );
    }

    if (Array.isArray(row["roles"])) {
      row["roles"] = (
        <UserCellDropdown
          options={[
            "Преподаватель",
            "Сотрудник",
            "Учебный офис",
            "Менеджер департамента",
            "Администратор",
          ]}
          defaultOptions={row["roles"]}
          tableNo={0}
          r={i}
          c={"roles"}
          userId={row["user_id"]}
        />
      );
    }

    row["authorization_date"] =
      new Date(row["authorization_date"]) != "Invalid Date"
        ? new Date(row["authorization_date"]).toLocaleDateString()
        : row["authorization_date"];

    row["last_login_date"] =
      new Date(row["last_login_date"]) != "Invalid Date"
        ? new Date(row["last_login_date"]).toLocaleDateString() +
          " " +
          new Date(row["last_login_date"]).toLocaleTimeString()
        : row["last_login_date"];

    row["token"] = <TokenPopup userId={row["user_id"]} />;
  });

  /**
   * Функция получения столбцов таблицы
   * @returns {array} Массив столбцов
   */
  const getColumns = () => {
    const resColumns = [];
    columns.map((column) => {
      if (!hiddenColumns.includes(column.accessor)) {
        resColumns.push(column);
      }
    });
    return resColumns;
  };

  /**
   * Функция удаления всех фильтров
   */
  const removeAllFilters = () => {
    dispatcher(toggleRemoveAllFilters());
  };

  return (
    <StaticPage>
      <div
        className={classNames(
          global_styles.white_rounded_box,
          styles.container
        )}
      >
        {data.length === 0 ? (
          <div className={styles.loading_wrapper}>
            <ThreeDots
              visible
              height="90"
              width="90"
              color="#FF8000"
              radius="9"
              ariaLabel="three-dots-loading"
              wrapperStyle={{}}
              wrapperClass=""
            />
          </div>
        ) : (
          <>
            <div className={styles.top}>
              <div className={styles.left_side}>
                <button
                  className={classNames(global_styles.btn_orange)}
                  onClick={removeAllFilters}
                >
                  Сбросить фильтры
                </button>
              </div>

              <div className={styles.right_side}>
                <DownloadTable
                  options={[".PDF", ".XLSX", ".CSV"]}
                  tableName={"users"}
                />
              </div>
            </div>

            <div className={styles.divider} />

            <Table
              columns={getColumns()}
              data={data}
              isSorted={true}
              isFiltered={true}
              className={"multi_table"}
            />
          </>
        )}
      </div>
    </StaticPage>
  );
};

export default Users;
