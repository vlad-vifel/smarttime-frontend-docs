import React, { useState, useEffect, useMemo } from "react";
import PropTypes from "prop-types";
import styles from "./SetDate.module.css";
import global_styles from "../../../../../styles/global.module.css";
import { DayPicker } from "react-day-picker";
import { useSelector } from "react-redux";
import { ru } from "date-fns/locale";
import classNames from "classnames";
import moment from "moment";
import { momentLocalizer } from "react-big-calendar";
import Popup from "../../../../common/popup/Popup";
import { patchExam } from "../requests";

moment.locale("ru");
const localizer = momentLocalizer(moment);

/**
 * Компонент выбора даты и времени начала и конца экзамена
 * @component
 */
const SetDate = ({
  start: initialStart,
  end: initialEnd,
  examId,
  number,
  year,
  toggleRerender,
}) => {
  const [isVisible, setIsVisible] = useState(false);
  const [range, setRange] = useState({});
  const [disabledDays, setDisabledDays] = useState([]);
  const [isSelected, setIsSelected] = useState(initialStart ? true : false);

  const [date, setDate] = useState(initialStart ? new Date(initialStart) : null);
  const [startDate, setStartDate] = useState(
    initialStart ? new Date(initialStart) : null
  );
  const [endDate, setEndDate] = useState(
    initialStart ? new Date(initialEnd) : null
  );

  const lessons = useSelector((state) => state.schedule.lessons);
  const modules = useSelector((state) => state.workload.modules);;

  const startDateId = useMemo(() => {
    return startDate
      ? lessons.find(
          (lesson) =>
            lesson.from.toLocaleTimeString().slice(0, 5) ==
            startDate.toLocaleTimeString().slice(0, 5)
        ).id
      : null;
  }, [startDate]);

  const endDateId = useMemo(
    () =>
      endDate
        ? lessons.find(
            (lesson) =>
              lesson.to.toLocaleTimeString().slice(0, 5) ==
              endDate.toLocaleTimeString().slice(0, 5)
          ).id
        : null,
    [endDate]
  );

  const optionsStart = useMemo(() => {
    const resOptions = [];
    lessons.forEach((lesson) => {
      resOptions.push(
        `${lesson.from.toLocaleTimeString().slice(0, 5)} (${lesson.id} пара)`
      );
    });
    return resOptions;
  }, [lessons]);

  const optionsEnd = useMemo(() => {
    const resOptions = [];
    lessons.forEach((lesson) => {
      resOptions.push(
        `${lesson.to.toLocaleTimeString().slice(0, 5)} (${lesson.id} пара)`
      );
    });
    return resOptions;
  }, [lessons]);

  useEffect(() => {
    setIsSelected(initialStart ? true : false);
    setDate(initialStart ? new Date(initialStart) : null);
    setStartDate(initialStart ? new Date(initialStart) : null);
    setEndDate(initialStart ? new Date(initialEnd) : null);
  }, [number, year, initialStart, initialEnd, examId]);

  useEffect(() => {
    const module = modules.find(
      (el) => el.year == year && el.number == number
    );
    const tmpRange = {
      from: new Date(module.session_start),
      to: new Date(module.session_end),
    };
    setRange(tmpRange);

    let current = tmpRange.from;
    const days = [];

    while (localizer.lte(current, tmpRange.to, "day")) {
      if (current.getDay() == 0) {
        days.push(current);
      }
      current = localizer.add(current, 1, "day");
    }

    setDisabledDays(days);
  }, [modules]);

  /**
   * Функция установки всех дат
   * @param {Date} newDate Новая дата
   */
  const setAllDates = (newDate) => {
    setDate(newDate);

    setStartDate(null);
    setEndDate(null);
  };

  /**
   * Функция установки времени начала экзамена
   * @param {number} lessonNr Номер пары
   */
  const setStart = (lessonNr) => {
    if (
      startDate &&
      startDate.getHours() === lessons[lessonNr].from.getHours() &&
      startDate.getMinutes() === lessons[lessonNr].from.getMinutes()
    ) {
      setStartDate(null);
    } else {
      const tmp = new Date(date.getTime());
      tmp.setHours(lessons[lessonNr].from.getHours());
      tmp.setMinutes(lessons[lessonNr].from.getMinutes());
      setStartDate(tmp);
    }
  };

  /**
   * Функция установки времени окончания экзамена
   * @param {number} lessonNr Номер пары
   */
  const setEnd = (lessonNr) => {
    if (
      endDate &&
      endDate.getHours() === lessons[lessonNr].to.getHours() &&
      endDate.getMinutes() === lessons[lessonNr].to.getMinutes()
    ) {
      setEndDate(null);
    } else {
      const tmp = new Date(date.getTime());
      tmp.setHours(lessons[lessonNr].to.getHours());
      tmp.setMinutes(lessons[lessonNr].to.getMinutes());
      setEndDate(tmp);
    }
  };

  /**
   * Функция переключения видимости всплывающего окна
   */
  const toggle = () => {
    setIsVisible(!isVisible);
  };

  /**
   * Функция cохранения выбранных дат и времени экзамена
   */
  const submit = () => {
    patchExam(examId, {
      start: startDate,
      end: endDate,
    }).then(() => {
      setTimeout(toggleRerender, 10);
      setIsSelected(true);
    });

    setIsVisible(false);
  };

  return (
    <>
      <button
        className={classNames(
          isSelected
            ? global_styles.btn_orange_reverse
            : global_styles.btn_orange,
          "centered"
        )}
        onClick={toggle}
      >
        {isSelected ? "Изменить" : "Назначить"}
      </button>
      <Popup
        headerTitle="Назначьте экзамен"
        isActive={isVisible}
        setActive={toggle}
        className={"datecalendar_nomargin"}
      >
        <div className={styles.main}>
          <div className={styles.top}>
            <div className={styles.wrapper}>
              <div className={styles.title}>Выберите дату</div>
              <DayPicker
                mode="single"
                onSelect={setAllDates}
                selected={date}
                showOutsideDays
                locale={ru}
                defaultMonth={range.from}
                fromDate={range.from}
                toDate={range.to}
                disabled={disabledDays}
              />
            </div>
            <div className={styles.wrapper}>
              <div className={styles.title}>Выберите время начала</div>
              <div className={styles.scroll}>
                {optionsStart.map((option, i) => (
                  <div key={i}>
                    <button
                      className={classNames(
                        styles.scroll_btn,
                        global_styles.btn_orange,
                        {
                          [styles.scroll_btn_selected]: startDate
                            ? startDateId == i + 1
                            : false,
                          [styles.scroll_btn_disabled]:
                            !date || (endDateId && endDateId < i + 1),
                        }
                      )}
                      onClick={() => setStart(i)}
                    >
                      {option}
                    </button>
                  </div>
                ))}
              </div>
            </div>

            <div className={styles.wrapper}>
              <div className={styles.title}>Выберите время конца</div>
              <div className={styles.scroll}>
                {optionsEnd.map((option, i) => (
                  <div key={i}>
                    <button
                      className={classNames(
                        styles.scroll_btn,
                        global_styles.btn_orange,
                        {
                          [styles.scroll_btn_selected]: endDate
                            ? endDateId == i + 1
                            : false,
                          [styles.scroll_btn_disabled]:
                            !date || startDateId > i + 1,
                        }
                      )}
                      onClick={() => setEnd(i)}
                    >
                      {option}
                    </button>
                  </div>
                ))}
              </div>
            </div>
          </div>

          <div className={styles.bottom}>
            <button
              className={classNames(global_styles.btn_gray)}
              onClick={() => setIsVisible(false)}
            >
              Отменить
            </button>

            <button
              className={classNames(global_styles.btn_orange)}
              onClick={submit}
              disabled={!(date && startDate && endDate)}
            >
              Сохранить
            </button>
          </div>
        </div>
      </Popup>
    </>
  );
};

SetDate.propTypes = {
  /**
   * Дата и время начала экзамена
   */
  start: PropTypes.string,
  /**
   * Дата и время окончания экзамена
   */
  end: PropTypes.string,
  /**
   * Идентификатор экзамена
   */
  examId: PropTypes.string.isRequired,
  /**
   * Номер модуля
   */
  number: PropTypes.number.isRequired,
  /**
   * Год
   */
  year: PropTypes.number.isRequired,
  /**
   * Функция перерендера компонента
   */
  toggleRerender: PropTypes.func.isRequired,
};

export default SetDate;
