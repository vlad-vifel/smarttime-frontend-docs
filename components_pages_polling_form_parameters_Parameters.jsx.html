import Table from "../../../../common/table/Table";
import styles from "./Parameters.module.css";
import "./Parameters.css";
import RadioButtons from "../../../../common/radiobuttons/RadioButtons";
import CellDropdown from "../cellDropdown/CellDropdown";
import Input from "../../../../common/input/Input";
import Question from "../../../../../assets/icons/question";
import { useEffect, useMemo, useState } from "react";
import { getForms } from "../requests";
import { useDispatch } from "react-redux";
import { setLocalData, setNoExams } from "../../../../../store/pollingSlice";
import { useLocation, useSearchParams } from "react-router-dom";
import { getModuleInfo } from "../../requests";

/**
 * Функция создания всплывающей подсказки
 * @param {string} text Текст, требующий подсказки
 * @param {string} tooltip Текст подсказки
 * @returns {JSX.Element} Всплывающая подсказка
 */
const makeTooltip = (text, tooltip) => {
  return (
    <div className={styles.tooltip_text}>
      {text}
      <div className={styles.tooltip_icon} title={tooltip}>
        <Question />
      </div>
    </div>
  );
};

/**
 * Компонент таблицы параметров экзаменов
 * @component
 */
const Parameters = () => {
  const dispatcher = useDispatch();

  const [data, setData] = useState();
  const [columns, setColumns] = useState();
  const [searchParams, setSearchParams] = useSearchParams();

  const module = useLocation().pathname.split("/").pop();

  const user = useMemo(
    () => (searchParams.get("user") ? `${searchParams.get("user")}` : null),
    [searchParams]
  );

  let groups = [];

  useEffect(() => {
    getModuleInfo(module).then((moduleInfo) => {
      getForms(moduleInfo.year, moduleInfo.number, user).then((forms) => {
        if (forms.length === 0) {
          dispatcher(setNoExams(true));
        } else {
          dispatcher(setNoExams(false));
        }

        groups = forms.map(
          (row) =>
            `${row["subject_name"]}, ${row["course"]} курс ${
              row["short_name"]
            }, ${row["groups"].join(", ")} (${row["units"].join(", ")})`
        );

        const tmpColumns = [
          {
            Header: "ㅤ",
            accessor: "field",
            width: "",
          },
        ];

        groups.forEach((group, i) => {
          tmpColumns.push({
            Header: group,
            accessor: "column" + (i + 1).toString(),
            width: "400",
          });
        });

        setColumns(tmpColumns);

        const row1 = {
          field: makeTooltip(
            "Формат проведения экзамена",
            "Выберите формат проведения экзамена из предложенных вариантов"
          ),
        };
        const row2 = {
          field: makeTooltip(
            "Контингент",
            "Выберите, проводится ли экзамен для группы или сразу для всего потока"
          ),
        };
        const row3 = {
          field: makeTooltip(
            "Максимальное количество групп в день",
            "Выберите максимальное количество групп, которое Вы сможете принять в один день"
          ),
        };
        const row4 = {
          field: makeTooltip(
            "Продолжительность экзамена",
            "Выберите необходимое количество пар для проведения экзамена"
          ),
        };
        const row5 = {
          field: makeTooltip(
            "Тип аудитории",
            "Выберите, какой тип аудитории необходим для проведения экзамена"
          ),
        };
        const row6 = {
          field: makeTooltip(
            "Комментарий",
            "В комментарии Вы можете указать дополнительные пожелания"
          ),
        };

        for (let i = 1; i <= groups.length; i++) {
          if (forms[i - 1]["editable"] == true) {
            if (forms[i - 1]["format"] === "Онлайн") {
              row1["column" + i.toString()] = "Онлайн";
            } else {
              row1["column" + i.toString()] = (
                <CellDropdown
                  options={["Письменный", "Устный", "Оценки по накопленной"]}
                  defaultOption={forms[i - 1]["format"]}
                  tableNo={0}
                  r={0}
                  c={"column" + i.toString()}
                  isScrollPage={true}
                  center={true}
                  isUpdate
                  name={"format"}
                />
              );
            }
            row2["column" + i.toString()] = (
              <RadioButtons
                options={["Поток", "Группа"]}
                defaultOption={forms[i - 1]["is_flow"] || "Группа"}
                c={"column" + i.toString()}
                isUpdate
                name={"is_flow"}
              />
            );
            row3["column" + i.toString()] = (
              <RadioButtons
                options={["1", "2", "3", "4", "5"]}
                defaultOption={forms[i - 1]["max_groups_count"] || "1"}
                c={"column" + i.toString()}
                isUpdate
                name={"max_groups_count"}
                flowDisable
              />
            );
            row4["column" + i.toString()] = (
              <CellDropdown
                options={[
                  "1 пара",
                  "2 пары",
                  "3 пары",
                  "4 пары",
                  "5 пар",
                  "6 пар",
                ]}
                defaultOption={forms[i - 1]["duration"]}
                tableNo={0}
                r={4}
                c={"column" + i.toString()}
                isScrollPage={true}
                center={true}
                isUpdate
                name={"duration"}
              />
            );
            if (forms[i - 1]["format"] === "Онлайн") {
              row5["column" + i.toString()] = "Онлайн";
            } else {
              row5["column" + i.toString()] = (
                <CellDropdown
                  options={[
                    "Лекционная",
                    "Семинарская",
                    "Компьютерный класс",
                  ]}
                  defaultOption={forms[i - 1]["room_type"]}
                  tableNo={0}
                  r={5}
                  c={"column" + i.toString()}
                  isScrollPage={true}
                  center={true}
                  isUpdate
                  name={"room_type"}
                />
              );
            }
            row6["column" + i.toString()] = (
              <Input
                hint="Написать комментарий"
                value={forms[i - 1]["comment"]}
                r={5}
                c={"column" + i.toString()}
                isUpdate
                name={"comment"}
              />
            );
          } else {
            row1["column" + i.toString()] = forms[i - 1]["format"] || "—";
            row2["column" + i.toString()] = forms[i - 1]["is_flow"] || "—";
            row3["column" + i.toString()] =
              forms[i - 1]["max_groups_count"] || "—";
            row4["column" + i.toString()] = forms[i - 1]["duration"] || "—";
            row5["column" + i.toString()] = forms[i - 1]["room_type"] || "—";
            row6["column" + i.toString()] = forms[i - 1]["comment"] || "—";
          }
        }

        const localData = [];

        for (let i = 1; i <= groups.length; i++) {
          localData.push({
            column: "column" + i.toString(),
            format: forms[i - 1]["format"],
            is_flow: forms[i - 1]["is_flow"] || "Группа",
            max_groups_count: forms[i - 1]["max_groups_count"] || "1",
            duration: forms[i - 1]["duration"],
            room_type: forms[i - 1]["room_type"], 
            comment: forms[i - 1]["comment"],
            unit_id: forms[i - 1]["unit_id"],
            editable: forms[i - 1]["editable"],
          });
        }

        dispatcher(setLocalData(localData));

        setData([row1, row2, row3, row4, row5, row6]);
      });
    });
  }, [searchParams]);

  return (
    <>
      {data && (
        <Table
          data={data}
          columns={columns}
          center={true}
          firstColumnSticky={true}
          className={"parameters_table"}
        />
      )}
    </>
  );
};

export default Parameters;
