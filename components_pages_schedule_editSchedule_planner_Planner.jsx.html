import classNames from "classnames";
import global_styles from "../../../../../styles/global.module.css";
import "react-big-calendar/lib/sass/styles.scss";
import "./Planner.css";
import styles from "./Planner.module.css";
import "react-big-calendar/lib/addons/dragAndDrop/styles.css";

import { Calendar, momentLocalizer } from "react-big-calendar";
import moment from "moment";
import "moment/locale/ru";
import { useCallback, useEffect, useMemo, useState } from "react";
import withDragAndDrop from "react-big-calendar/lib/addons/dragAndDrop";
import "react-big-calendar/lib/addons/dragAndDrop/styles.css";

import Globalize from "globalize";
import MyRange from "./myRange/MyRange";
import MyEventComponent from "./myEvent/MyEventComponent";
import MyEventWrapperComponentDraggable from "./myEventWrapper/MyEventWrapperComponentDraggable";
import { useDispatch, useSelector } from "react-redux";
import {
  setEvents,
  setSelectedEvent,
  updateEvent,
} from "../../../../../store/sheduleSlice";

import { toRuFormat } from "../../../../../services/helper";
import { useLocation } from "react-router-dom";
import { dropdowns } from "../tools/Tools";
import { getExams } from "../requests";
import AlertWrongDate from "../popups/AlertWrongDate";
import AlertBusyAuditorium from "../popups/AlertBusyAuditorium";
import AlertUnexeptableGroupTime from "../popups/AlertUnexeptableGroupTime";
import AlertUnexeptableTeacherTime from "../popups/AlertUnexeptableTeacherTime";

Globalize.culture("ru-RU");
moment.locale("ru");
const localizer = momentLocalizer(moment);
const lang = {
  en: null,
  "ru-RU": {
    week: "Неделя",
    work_week: "Рабочая неделя",
    day: "День",
    month: "Месяц",
    previous: "Предыдущая",
    next: "Следующая",
    today: "Сегодня",
    agenda: "Дневник",

    showMore: (total) => `+${total} больше`,
  },
};

const plannerModes = [
  {
    step: 10,
    timeslots: 3,
  },
  {
    step: 10,
    timeslots: 6,
  },
];

const startHour = new Date(0, 0, 0, 9, 0, 0);
const endHour = new Date(0, 0, 0, 21, 0, 0);

const DnDCalendar = withDragAndDrop(Calendar);

const updateThisEvent = (allEvents, updatedEvent) => {
  const filtered = allEvents.filter((item) => item.id !== updatedEvent.id);

  return [...filtered, { ...updatedEvent }];
};

/**
 * Компонент календаря с расписанием
 * @component
 */
const Planner = () => {
  const initialEvents = useSelector((state) => state.schedule.events);
  const allFilters = useSelector((state) => state.schedule.selectedFilters);
  const plannerMode = useSelector((state) => state.schedule.plannerMode);
  const lessons = useSelector((state) => state.schedule.lessons);
  const defaultRange = useSelector((state) => state.schedule.defaultRange);
  const disabledEdit = useSelector((state) => state.schedule.disabledEdit);
  const module = useLocation().pathname.split("/").pop();

  const dispatcher = useDispatch();

  const [showEvents, setShowEvents] = useState(initialEvents);
  const [culture, setCulture] = useState("ru-RU");
  const [isVisibleWrongDate, setIsVisibleWrongDate] = useState(false);
  const [isVisibleBusyAuditorium, setIsVisibleBusyAuditorium] = useState(false);
  const [isVisibleUnexeptableGroupTime, setIsVisibleUnexeptableGroupTime] =
    useState(false);
  const [isVisibleUnexeptableTeacherTime, setIsVisibleUnexeptableTeacherTime] =
    useState(false);

  const rbc_header_buttons = document.getElementsByClassName("rbc-header");

  useEffect(() => {
    Array.from(rbc_header_buttons).forEach((header) => {
      header.getElementsByTagName("button")[0].onclick = null;
    });
  }, []);

  useEffect(() => {
    getExams(module).then((exams) => dispatcher(setEvents(exams)));
  }, []);

  const onEventDrop = useCallback(
    ({ event, start, end, isAllDay: droppedOnAllDaySlot = false }) => {
      if (!event.allDay && droppedOnAllDaySlot) {
        event.allDay = true;
      }

      let newStart;
      let newEnd;
      let minDiffStart = Infinity;
      let minDiffEnd = Infinity;
      lessons.forEach((lesson) => {
        const diffStart = Math.abs(
          new Date(0, 0, 0, start.getHours(), start.getMinutes()) - lesson.from
        );
        const diffEnd = Math.abs(
          new Date(0, 0, 0, end.getHours(), end.getMinutes()) - lesson.to
        );
        if (diffStart < minDiffStart) {
          newStart = lesson.from;
          minDiffStart = diffStart;
        }
        if (diffEnd < minDiffEnd) {
          newEnd = lesson.to;
          minDiffEnd = diffEnd;
        }
      });

      if (minDiffStart !== Infinity && minDiffEnd !== Infinity) {
        start.setHours(newStart.getHours());
        start.setMinutes(newStart.getMinutes());
        end.setHours(newEnd.getHours());
        end.setMinutes(newEnd.getMinutes());
      }

      // Проверка на допустимые значения
      const possibleEvent = { ...event, start, end };
      let isBusyAuditorium = false;
      let isUnexeptableGroupTime = false;
      let isUnexeptableTeacherTime = false;

      initialEvents.forEach((event) => {
        if (
          event.id !== possibleEvent.id &&
          event.start <= possibleEvent.end &&
          event.end >= possibleEvent.start &&
          possibleEvent.format !== "Оценки по накопленной" &&
          event.format !== "Оценки по накопленной"
        ) {
          if (
            event.auditorium == possibleEvent.auditorium &&
            event.format !== "Онлайн"
          ) {
            isBusyAuditorium = true;
          }
          possibleEvent.groups.forEach((group) => {
            if (event.groups.includes(group)) {
              isUnexeptableGroupTime = true;
            }
          });

          possibleEvent.teachers.forEach((teacher) => {
            if (event.teachers.includes(teacher)) {
              isUnexeptableTeacherTime = true;
            }
          });
        }
      });

      if (possibleEvent.start.getDay() == 0) {
        setIsVisibleWrongDate(true);
      } else if (isBusyAuditorium) {
        setIsVisibleBusyAuditorium(true);
      } else if (isUnexeptableGroupTime) {
        setIsVisibleUnexeptableGroupTime(true);
      } else if (isUnexeptableTeacherTime) {
        setIsVisibleUnexeptableTeacherTime(true);
      } else {
        setShowEvents((prev) => updateThisEvent(prev, possibleEvent));
        dispatcher(setSelectedEvent(possibleEvent));
        dispatcher(updateEvent(possibleEvent));
      }
    },
    [initialEvents]
  );

  const onEventResize = useCallback(
    ({ event, start, end }) => {
      if (event.start.toLocaleTimeString() !== start.toLocaleTimeString()) {
        let newStart;
        let minDiff = Infinity;
        lessons.forEach((lesson) => {
          const diff = Math.abs(
            new Date(0, 0, 0, start.getHours(), start.getMinutes()) -
              lesson.from
          );
          if (
            lesson.from < new Date(0, 0, 0, end.getHours(), end.getMinutes()) &&
            diff < minDiff
          ) {
            newStart = lesson.from;
            minDiff = diff;
          }
        });
        start.setHours(newStart.getHours());
        start.setMinutes(newStart.getMinutes());
      } else if (event.end.toLocaleTimeString() !== end.toLocaleTimeString()) {
        let newEnd;
        let minDiff = Infinity;
        lessons.forEach((lesson) => {
          const diff = Math.abs(
            new Date(0, 0, 0, end.getHours(), end.getMinutes()) - lesson.to
          );
          if (
            lesson.to >
              new Date(0, 0, 0, start.getHours(), start.getMinutes()) &&
            diff < minDiff
          ) {
            newEnd = lesson.to;
            minDiff = diff;
          }
        });
        end.setHours(newEnd.getHours());
        end.setMinutes(newEnd.getMinutes());
      }

      // Проверка на допустимые значения
      const possibleEvent = { ...event, start, end };
      let isBusyAuditorium = false;
      let isUnexeptableGroupTime = false;
      let isUnexeptableTeacherTime = false;

      initialEvents.forEach((event) => {
        if (
          event.id !== possibleEvent.id &&
          event.start <= possibleEvent.end &&
          event.end >= possibleEvent.start &&
          possibleEvent.format !== "Оценки по накопленной" &&
          event.format !== "Оценки по накопленной"
        ) {
          if (
            event.auditorium == possibleEvent.auditorium &&
            event.format !== "Онлайн"
          ) {
            isBusyAuditorium = true;
          }
          possibleEvent.groups.forEach((group) => {
            if (event.groups.includes(group)) {
              isUnexeptableGroupTime = true;
            }
          });

          possibleEvent.teachers.forEach((teacher) => {
            if (event.teachers.includes(teacher)) {
              isUnexeptableTeacherTime = true;
            }
          });
        }
      });

      if (isBusyAuditorium) {
        setIsVisibleBusyAuditorium(true);
      } else if (isUnexeptableGroupTime) {
        setIsVisibleUnexeptableGroupTime(true);
      } else if (isUnexeptableTeacherTime) {
        setIsVisibleUnexeptableTeacherTime(true);
      } else {
        setShowEvents((prev) => updateThisEvent(prev, possibleEvent));
        dispatcher(setSelectedEvent(possibleEvent));
        dispatcher(updateEvent(possibleEvent));
      }
    },
    [initialEvents]
  );

  useEffect(() => {
    let resEvents = [...initialEvents];

    dropdowns.forEach((dropdown) => {
      const id = dropdown.id;
      const selectedValues =
        allFilters.find((filter) => filter.id == id)?.selectedValues || [];
      if (selectedValues.length > 0) {
        if (dropdown.inArray) {
          resEvents = resEvents.filter((event) => {
            let summ = 0;
            event[id].forEach((value) => {
              if (selectedValues.includes(`${value}`)) {
                summ += 1;
              }
            });
            return summ > 0;
          });
        } else {
          resEvents = resEvents.filter((event) =>
            selectedValues.includes(`${event[id]}`)
          );
        }
      }
    });
    setShowEvents(resEvents);
  }, [allFilters, initialEvents]);

  const { defaultDate, formats, messages, views, events, components } = useMemo(
    () => ({
      // defaultDate: new Date(2024, 3, 22),
      formats: {
        // the day of the week header in the 'month' view
        weekdayFormat: (date, culture, localizer) =>
          localizer.format(date, "dddd", culture),
        // the day header in the 'week' and 'day' (Time Grid) views
        dayFormat: (date, culture, localizer) =>
          toRuFormat(localizer.format(date, "D MMM ddd", culture)),
        // the time in the gutter in the Time Grid views
        timeGutterFormat: (date, culture, localizer) =>
          localizer.format(date, "HH:mm", culture),
        eventTimeRangeFormat: ({ start, end }, culture, localizer) =>
          localizer.format(start, "HH:mm", culture) +
          " - " +
          localizer.format(end, "HH:mm", culture),
      },
      messages: lang[culture],
      views: {
        week: MyRange,
      },
      events: showEvents,
      components: {
        event: MyEventComponent,
        eventWrapper: MyEventWrapperComponentDraggable,
      },
    }),
    [culture, initialEvents, showEvents]
  );

  const onSelectEvent = useCallback((calEvent) => {
    calEvent.start = calEvent.start.toISOString();
    calEvent.end = calEvent.end.toISOString();
    dispatcher(setSelectedEvent(calEvent));
  }, []);

  return (
    <div
      className={classNames(
        global_styles.white_rounded_box,
        styles.planner_big,
        "planner"
      )}
      style={{ "--n": "2" }}
    >
      {Object.keys(defaultRange).length !== 0 && (
        <DnDCalendar
          localizer={localizer}
          events={events}
          defaultView={"week"}
          toolbar={false}
          defaultDate={defaultDate}
          formats={formats}
          culture="ru-RU"
          messages={messages}
          views={views}
          onSelectEvent={onSelectEvent}
          components={components}
          draggableAccessor={disabledEdit}
          popup
          resizable
          onEventDrop={onEventDrop}
          onEventResize={onEventResize}
          min={startHour}
          max={endHour}
          step={plannerModes[plannerMode].step}
          timeslots={plannerModes[plannerMode].timeslots}
        />
      )}
      <AlertWrongDate
        isActive={isVisibleWrongDate}
        setActive={setIsVisibleWrongDate}
        btnClick1={() => setIsVisibleWrongDate(false)}
      />
      <AlertBusyAuditorium
        isActive={isVisibleBusyAuditorium}
        setActive={setIsVisibleBusyAuditorium}
        btnClick1={() => setIsVisibleBusyAuditorium(false)}
      />
      <AlertUnexeptableGroupTime
        isActive={isVisibleUnexeptableGroupTime}
        setActive={setIsVisibleUnexeptableGroupTime}
        btnClick1={() => setIsVisibleUnexeptableGroupTime(false)}
      />
      <AlertUnexeptableTeacherTime
        isActive={isVisibleUnexeptableTeacherTime}
        setActive={setIsVisibleUnexeptableTeacherTime}
        btnClick1={() => setIsVisibleUnexeptableTeacherTime(false)}
      />
    </div>
  );
};

export default Planner;
