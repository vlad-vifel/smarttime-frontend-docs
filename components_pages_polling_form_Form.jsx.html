components/pages/polling/form/Form.jsx

import classNames from "classnames";
import styles from "./Form.module.css";

import global_styles from "../../../../styles/global.module.css";
import ScrollablePage from "../../common/scrollablePage/ScrollablePage";
import { useSelector } from "react-redux";
import TeacherDropdown from "./teacherDropdown/TeacherDropdown";
import Collapsible from "./collapsible/Collapsible";
import Restrictions from "./restrictions/Resctrictions";
import Parameters from "./parameters/Parameters";
import { postForm } from "./requests";
import CheckboxRestrictions from "./restrictions/checkboxRestrictions/CheckboxRestrictions";
import { useLocation, useSearchParams } from "react-router-dom";
import { useEffect, useMemo, useState } from "react";
import { getModuleInfo } from "../requests";
import Popup from "../../../common/popup/Popup";
import { getUserInfo } from "../../../../services/request";

/**
 * Компонент страницы Анкета
 * @component
 */
const Form = () => {
  const [moduleInfo, setModuleInfo] = useState({});
  const [userInfo, setUserInfo] = useState({});
  const [isVisible, setIsVisible] = useState(false);
  const [header, setHeader] = useState("Завершение анкеты");
  const [text, setText] = useState("Анкета успешно сохранена");

  const [searchParams, setSearchParams] = useSearchParams();
  const user = useMemo(
    () => (searchParams.get("user") ? `${searchParams.get("user")}` : null),
    [searchParams]
  );

  const formats = useSelector((state) => state.polling.localData);
  const restrictions = useSelector((state) => state.polling.restrictions);
  const noExams = useSelector((state) => state.polling.noExams);
  const module = useLocation().pathname.split("/").pop();
  const selectedRole = localStorage.getItem("selectedRole");

  useEffect(() => {
    getModuleInfo(module).then((info) => setModuleInfo(info));
    getUserInfo().then((info) => setUserInfo(info));
  }, []);

  /**
   * Функция переключения видимости всплывающего окна
   */
  const toggle = () => {
    setIsVisible(!isVisible);
  };

  /**
   * Функция проверки заполнености всех необходимых параметров экзамена
   * @param {array} formats Параметры экзаменов
   * @returns {boolean} Флаг, показывающий, заполнены ли все необходимые параметры экзамена
   */
  const checkFullFilled = (formats) => {
    let cnt = 0;
    formats.forEach((format) => {
      if (
        format.editable &&
        !(
          format.unit_id &&
          format.format &&
          format.duration &&
          format.room_type
        )
      ) {
        cnt += 1;
      }
    });
    return cnt === 0;
  };

  /**
   * Функция проверки на достаточное количество свободного времени у преподавателя для проведения экзаменов
   * @param {array} formats Параметры экзаменов
   * @param {object} restrictions Ограничения преподавателя
   * @returns {boolean} Флаг, показывающий, достаточно ли свободного времени у преподавателя для проведения экзаменов
   */
  const checkEnoughTime = (formats, restrictions) => {
    let allExamDuration = 0;
    formats.forEach((el) => {
      allExamDuration += el.duration ? Number(el.duration.slice(0, 1)) : 0;
    });

    const allDays =
      Object.keys(restrictions).length - restrictions["disabled"].length - 1;
    let uncomfortablePairs = 0;
    Object.keys(restrictions).forEach((key) => {
      if (key !== "disabled") {
        if (restrictions[key].includes(0)) {
          uncomfortablePairs += 7;
        } else {
          uncomfortablePairs += restrictions[key].length;
        }
      }
    });

    const availableExamDuration = allDays * 7 - uncomfortablePairs;

    return availableExamDuration >= 2 * allExamDuration;
  };

  /**
   * Функция сохранения и отправки анкеты
   */
  const saveForm = () => {
    const isFullFilled = checkFullFilled(formats);
    const isEnoughTime = checkEnoughTime(formats, restrictions);
    if (!isFullFilled) {
      setHeader("Предупреждение");
      setText("Вы заполнили не все поля");
      setIsVisible(true);
    } else if (!isEnoughTime) {
      setHeader("Предупреждение");
      setText(
        "Количество пар, в которое Вы можете принимать экзамен, должно быть как минимум в 2 раза больше общей продолжительности всех Ваших экзаменов"
      );
      setIsVisible(true);
    } else {
      const { disabled, ...newRestrictions } = restrictions;
      postForm(
        formats,
        newRestrictions,
        moduleInfo.year,
        moduleInfo.number,
        user
      ).then((data) => {
        if (data.formats.success && data.excluded_schedule.success) {
          setHeader("Завершение анкеты");
          setText("Анкета успешно сохранена");
        } else {
          setHeader("Ошибка");
          setText("Произошла ошибка");
        }
        setIsVisible(true);
      });
    }
  };

  return (
    <ScrollablePage>
      <div className={styles.content}>
        <div
          className={classNames(global_styles.white_rounded_box, styles.intro)}
        >
          <div className={styles.intro_header}>
            <div>Здравствуйте,</div>
            {selectedRole === "Преподаватель" ? (
              <div className={styles.fio_container}>{userInfo.fio}</div>
            ) : (
              <TeacherDropdown />
            )}
          </div>
          <div className={styles.intro_text}>
            <p>
              В данной форме просим Вас сообщить о своих ограничениях, которые
              необходимо учесть при составлении расписания сессии{" "}
              {moduleInfo.number} модуля {moduleInfo.year}/{moduleInfo.year + 1}{" "}
              учебного года. Экзамены пройдут{" "}
              <span>
                c {new Date(moduleInfo.session_start).toLocaleDateString()} по{" "}
                {new Date(moduleInfo.session_end).toLocaleDateString()}.
              </span>{" "}
              В форму будут включены все экзамены в соответствии с Вашей учебной
              нагрузкой.
            </p>
            <p>
              Обращаем Ваше внимание, что направленные заявки в данной форме
              являются приоритетными по составлению расписания в сравнении с
              другими видами обращений.
            </p>
            <p>
              Сбор пожеланий осуществляется{" "}
              <span>
                до {new Date(moduleInfo.survey_end).toLocaleDateString()} до
                23:59
              </span>
              .
            </p>
          </div>
        </div>

        <Collapsible label="Инструкция" isOpen={true}>
          <div className={styles.instruction}>
            <p>
              В самом начале анкетирования расположена таблица с днями сессии по
              горизонтали и временами пар по вертикали. Она заполняется{" "}
              <span>для всех дисциплин сразу</span>. Вам необходимо выбрать
              время, когда Вам <span>не удобно</span> принимать экзамены. Также
              можно выбрать целый день, когда нет возможности принимать
              экзамены. Просьба делать выбор обдуманно без использования
              максимального количества ограничений.
            </p>
            <br />
            <p>
              Следующая анкета включает в себя{" "}
              <span>
                опросную форму отдельно по каждой дисциплине для каждой
                образовательной программы
              </span>
              . Необходимо выбрать только один вариант ответа для каждого пункта
              вопросов:
            </p>
            <ol>
              <li>
                Формат проведения экзамена:
                <ul>
                  <li>Письменный,</li>
                  <li>Устный,</li>
                  <li>Экзамена нет — выставление оценки по накопленной,</li>
                  <li>
                    Онлайн (для онлайн-программ, безопасности жизнедеятельности
                    и дисциплин общего цикла).
                  </li>
                </ul>
              </li>
              <li>
                Контингент (выбор формат экзамена по численности студентов):
                <ul>
                  <li>Поток,</li>
                  <li>Группа.</li>
                </ul>
              </li>
              <li>
                Максимальное количество групп в день: от 2 до 5 – выбор
                предпочтительного количества групп в день, у которых будет
                приниматься экзамен.
              </li>
              <li>
                Продолжительность экзамена (у группы/потока в зависимости от
                контингента): от 1 до 6 пар – выбор необходимой длительности для
                приема экзамена.
              </li>
              <li>Тип аудитории:</li>
              <ul>
                <li>Лекционная,</li>
                <li>Семинарская,</li>
                <li>Компьютерный класс,</li>
                <li>Учебная лаборатория,</li>
                <li>
                  Онлайн (для онлайн-программ, безопасности жизнедеятельности и
                  дисциплин общего цикла).
                </li>
              </ul>
              <li>При необходимости можно оставить комментарий к анкете.</li>
            </ol>
            <br />
            <p>Кнопка «Сохранить» сохранит Вашу анкету.</p>
            <br />
            <p>
              При необходимости <span>редактирования</span> анкеты появляется
              заполненная ранее форма. Если Вы хотите изменить свой выбор, то
              необходимо внести данные, а затем нажать кнопку «Сохранить». Новый
              ответ будет записан. Редактировать анкету можно{" "}
              <span>до окончания срока заполнения</span> (до{" "}
              {new Date(moduleInfo.survey_end).toLocaleDateString()}{" "}
              включительно).
            </p>
            <br />
            <p>Благодарим за сотрудничество!</p>
          </div>
        </Collapsible>

        {(selectedRole === "Преподаватель" || user) && (
          <div
            className={classNames(styles.wrapper, {
              [styles.hidden]: noExams,
            })}
          >
            <Collapsible label="Параметры для групп" isOpen={true}>
              <Parameters />
            </Collapsible>
            <Collapsible
              label="Ограничения для расписания сессии"
              isOpen={true}
            >
              <div className={styles.restrictions}>
                <p>
                  Отметьте в таблице время/дни, в которые Вы{" "}
                  <span>НЕ МОЖЕТЕ</span> принимать экзамены.
                </p>
                <div className={styles.checkboxes}>
                  <div>
                    <CheckboxRestrictions
                      checked={false}
                      disabled={true}
                      isClose={true}
                    />{" "}
                    - могу
                  </div>
                  <div>
                    <CheckboxRestrictions
                      checked={true}
                      disabled={true}
                      isClose={true}
                    />{" "}
                    - не могу
                  </div>
                </div>
                <Restrictions />
              </div>
            </Collapsible>
            <div
              className={classNames(
                styles.save_container,
                global_styles.white_rounded_box
              )}
            >
              <button className={global_styles.btn_orange} onClick={saveForm}>
                Сохранить
              </button>

              {isVisible && (
                <Popup
                  headerTitle={header}
                  isActive={isVisible}
                  setActive={toggle}
                >
                  <div className={styles.popup_body}>
                    <div>{text}</div>
                    <button
                      className={classNames(global_styles.btn_orange)}
                      onClick={toggle}
                    >
                      Продолжить
                    </button>
                  </div>
                </Popup>
              )}
            </div>
          </div>
        )}

        {noExams && (
          <div className={styles.no_exams}>
            У Вас нет экзаменов в этом модуле
          </div>
        )}
      </div>
    </ScrollablePage>
  );
};

export default Form;
